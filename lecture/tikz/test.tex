  \begin{framed}
    \setlength{\fboxrule}{2pt}
    \setlength{\fboxsep}{10pt}
    \setlength{\FrameSep}{20pt}
    % \setlength{\cornersize}{0.3cm}

  \end{framed}
}

\begin{flushleft}
\textbf{Example 1.} Find the value of $x$ if $2x + 5 = 13$.

\bigskip

\solutionbox{
  \begin{align*}
  2x + 5 &= 13 \\
  2x &= 8 \\
  x &= 4
  \end{align*}
}

\end{flushleft}
